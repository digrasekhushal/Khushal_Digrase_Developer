Greater Number


    let A = document.getElementById("input1");
    let B = document.getElementById("input2");
    let display = document.getElementById("display");

    function compareButton() {
        let num1 = Number(A.value);
        let num2 = Number(B.value);

        if (num1 > num2) {
            display.innerHTML = num1 + " Is greater than " + num2;
        } else if (num1 < num2) {
            display.innerHTML = num2 + " Is grester than " + num1;
        } else {
            display.innerHTML = "Both numbers are equal";
        }
    }
